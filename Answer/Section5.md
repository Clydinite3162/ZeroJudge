# find the median (中位數)
l = len(address)
if l % 2 == 0:
    median = (address[l//2] + address[l//2 - 1]) / 2
    if address[l//2] - median == median - address[l//2 - 1]:
        print(address[l//2 - 1])
    else:
        if address[l//2] - median > median - address[l//2 - 1]:
            print(address[l//2 - 1])
        else:
            print(address[l//2])
else:
    median = address[l//2]
    print(median)
